Manejo de Archivos en C y C++

El manejo de archivos en C y C++ permite a los programas leer y escribir datos en archivos. En C, se utilizan funciones de la biblioteca estándar stdio.h, mientras que en C++ se utilizan clases de la biblioteca fstream.

En C

En C, se utilizan las funciones de la biblioteca estándar stdio.h para realizar operaciones básicas de manejo de archivos. Estas operaciones incluyen abrir, leer, escribir y cerrar archivos.

Operaciones Básicas

                
                    #include <stdio.h>
            
                    int main() {
                        // Abrir un archivo para escritura
                        FILE *archivo = fopen("archivo.txt", "w");
            
                        if (archivo == NULL) {
                            perror("Error al abrir el archivo");
                            return 1;
                        }
            
                        // Escribir en el archivo
                        fprintf(archivo, "Hola, mundo!\n");
            
                        // Cerrar el archivo
                        fclose(archivo);
            
                        // Abrir un archivo para lectura
                        archivo = fopen("archivo.txt", "r");
            
                        if (archivo == NULL) {
                            perror("Error al abrir el archivo");
                            return 1;
                        }
            
                        char buffer[100];
                        // Leer del archivo
                        while (fgets(buffer, sizeof(buffer), archivo) != NULL) {
                            printf("%s", buffer);
                        }
            
                        // Cerrar el archivo
                        fclose(archivo);
            
                        return 0;
                    }

Funciones Importantes

                
                    // Abrir un archivo
                    FILE *fopen(const char *nombre_archivo, const char *modo);
            
                    // Leer de un archivo
                    size_t fread(void *ptr, size_t size, size_t nmemb, FILE *stream);
                    char *fgets(char *str, int n, FILE *stream);
            
                    // Escribir en un archivo
                    size_t fwrite(const void *ptr, size_t size, size_t nmemb, FILE *stream);
                    int fprintf(FILE *stream, const char *format, ...);
            
                    // Cerrar un archivo
                    int fclose(FILE *stream);

En C++

En C++, se utilizan las clases de la biblioteca estándar fstream para manejar archivos. Estas clases proporcionan una interfaz orientada a objetos para operaciones de entrada y salida de archivos.

Operaciones Básicas

                
                    #include <iostream>
                    #include <fstream>
            
                    int main() {
                        // Abrir un archivo para escritura
                        std::ofstream archivo("archivo.txt");
            
                        if (!archivo) {
                            std::cerr << "Error al abrir el archivo para escritura" << std::endl;
                            return 1;
                        }
            
                        // Escribir en el archivo
                        archivo << "Hola, mundo!" << std::endl;
            
                        // Cerrar el archivo
                        archivo.close();
            
                        // Abrir un archivo para lectura
                        std::ifstream archivo_lectura("archivo.txt");
            
                        if (!archivo_lectura) {
                            std::cerr << "Error al abrir el archivo para lectura" << std::endl;
                            return 1;
                        }
            
                        std::string linea;
                        // Leer del archivo
                        while (std::getline(archivo_lectura, linea)) {
                            std::cout << linea << std::endl;
                        }
            
                        // Cerrar el archivo
                        archivo_lectura.close();
            
                        return 0;
                    }

Clases Importantes

                
                    // Clase para escritura
                    std::ofstream archivo; // Abre el archivo para escritura
                    archivo.open("archivo.txt"); // Abrir archivo
                    archivo << "Texto"; // Escribir en el archivo
                    archivo.close(); // Cerrar el archivo
            
                    // Clase para lectura
                    std::ifstream archivo_lectura; // Abre el archivo para lectura
                    archivo_lectura.open("archivo.txt"); // Abrir archivo
                    std::string linea;
                    while (std::getline(archivo_lectura, linea)) {
                        std::cout << linea << std::endl; // Leer línea por línea
                    }
                    archivo_lectura.close(); // Cerrar el archivo
