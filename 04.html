Manejo de Archivos en C y C++

El manejo de archivos permite a los programas leer y escribir datos en archivos. En C y C++, esto se realiza mediante funciones y clases específicas para entrada/salida de archivos.

En C

En C, se utilizan las funciones de la biblioteca estándar stdio.h para manejar archivos. Estas funciones permiten abrir, leer, escribir y cerrar archivos.

                
                    #include <stdio.h>
            
                    int main() {
                        // Abrir un archivo para escritura
                        FILE *archivo = fopen("archivo.txt", "w");
            
                        if (archivo == NULL) {
                            perror("Error al abrir el archivo");
                            return 1;
                        }
            
                        // Escribir en el archivo
                        fprintf(archivo, "Hola, mundo!\n");
            
                        // Cerrar el archivo
                        fclose(archivo);
            
                        // Abrir un archivo para lectura
                        archivo = fopen("archivo.txt", "r");
            
                        if (archivo == NULL) {
                            perror("Error al abrir el archivo");
                            return 1;
                        }
            
                        char buffer[100];
                        // Leer del archivo
                        while (fgets(buffer, sizeof(buffer), archivo) != NULL) {
                            printf("%s", buffer);
                        }
            
                        // Cerrar el archivo
                        fclose(archivo);
            
                        return 0;
                    }

En C++

En C++, se utilizan las clases de la biblioteca estándar fstream para manejar archivos. Estas clases permiten un manejo más orientado a objetos de la entrada/salida de archivos.

                
                    #include <iostream>
                    #include <fstream>
            
                    int main() {
                        // Abrir un archivo para escritura
                        std::ofstream archivo("archivo.txt");
            
                        if (!archivo) {
                            std::cerr << "Error al abrir el archivo para escritura" << std::endl;
                            return 1;
                        }
            
                        // Escribir en el archivo
                        archivo << "Hola, mundo!" << std::endl;
            
                        // Cerrar el archivo
                        archivo.close();
            
                        // Abrir un archivo para lectura
                        std::ifstream archivo_lectura("archivo.txt");
            
                        if (!archivo_lectura) {
                            std::cerr << "Error al abrir el archivo para lectura" << std::endl;
                            return 1;
                        }
            
                        std::string linea;
                        // Leer del archivo
                        while (std::getline(archivo_lectura, linea)) {
                            std::cout << linea << std::endl;
                        }
            
                        // Cerrar el archivo
                        archivo_lectura.close();
            
                        return 0;
                    }
